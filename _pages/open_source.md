## Open Source Artifacts

<div class="artifact-page">

<p class="artifact-intro">
Besides peer-reviewed research publications, our group also makes our contributions available as open-source artifacts.
Although not required by the publication venues, we make these artifacts available at the time of publication of our regular
peer-reviewed research publications. The goals behind making these contributions open-source are threefold: to expedite
research progress as a community, lower the barrier to entry to new fields, and promote research reproducibility.
</p>

<script>
document.addEventListener("DOMContentLoaded", function () {
  document.querySelectorAll(".artifact-stats[data-zenodo-id]").forEach(function (el) {
    const id = el.getAttribute("data-zenodo-id");

    fetch(`https://zenodo.org/api/records/${id}`)
      .then(function (response) {
        if (!response.ok) {
          throw new Error("Zenodo API request failed");
        }
        return response.json();
      })
      .then(function (data) {
        const stats = data.stats || {};

        const views =
          stats.unique_views ??
          stats.version_unique_views ??
          stats.views ??
          "--";

        const downloads =
          stats.unique_downloads ??
          stats.version_unique_downloads ??
          stats.downloads ??
          "--";

        el.querySelector(".artifact-views").textContent = `Views: ${Number(views).toLocaleString()}`;
        el.querySelector(".artifact-downloads").textContent = `Downloads: ${Number(downloads).toLocaleString()}`;
      })
      .catch(function () {
        el.style.display = "none";
      });
  });
});
</script>
